import 'package:flutter/material.dart';
import 'package:slide_to_act/slide_to_act.dart';
import 'package:intl/intl.dart';

class TodayScreen extends StatefulWidget {
  const TodayScreen({Key? key}) : super(key: key);

  @override
  _TodayScreenState createState() => _TodayScreenState();
}

class _TodayScreenState extends State<TodayScreen> {
  String employeeId = 'E123'; // Employee ID
  String checkInTime = '--:--'; // Initial check-in time
  String checkOutTime = '--:--'; // Initial check-out time
  bool isCheckedIn = false; // Check-in/check-out status
  String currentDateTime = ''; // Current date and time
  String workStatus = '----'; // Initial work status

  @override
  void initState() {
    super.initState();
    currentDateTime = _getCurrentDateTime(); // Get current date and time
  }

  Future<void> toggleCheckInOut() async {
    if (!isCheckedIn) {
      // If user is checking in
      checkInTime = _getCurrentTime();
      checkOutTime = '--:--'; // Reset check-out time
      isCheckedIn = true; // Update status to checked in
      workStatus = _getWorkStatus(); // Update work status
    } else {
      // If user is checking out
      if (checkInTime == '--:--') {
        final snackBar = SnackBar(
          content: Text('You must check in before checking out!'),
          duration: const Duration(seconds: 2),
        );
        ScaffoldMessenger.of(context).showSnackBar(snackBar);
        return;
      }
      checkOutTime = _getCurrentTime();
      isCheckedIn = false; // Update status to checked out
    }

    setState(() {});

    // Show a SnackBar after check in/out
    final snackBar = SnackBar(
      content: Text(isCheckedIn ? 'Checked In at $checkInTime' : 'Checked Out at $checkOutTime'),
      duration: const Duration(seconds: 2),
    );
    ScaffoldMessenger.of(context).showSnackBar(snackBar);
  }

  String _getCurrentTime() {
    final now = DateTime.now();
    return '${now.hour.toString().padLeft(2, '0')}:${now.minute.toString().padLeft(2, '0')}';
  }

  String _getCurrentDateTime() {
    final now = DateTime.now();
    return DateFormat('dd/MM/yyyy').format(now); // Date format
  }

  String _getWorkStatus() {
    final checkInHour = DateTime.now().hour;
    final checkInMinute = DateTime.now().minute;

    // Check if arrived before or after 8:30
    if (checkInHour > 7 || (checkInHour == 8 && checkInMinute < 30)) {
      return 'ปกติ'; // On time
    } else {
      return 'สาย'; // Late
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Colors.white,
      body: SingleChildScrollView(
        child: Padding(
          padding: const EdgeInsets.all(16.0),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              const SizedBox(height: 20),
              // Employee Container
              Container(
                padding: const EdgeInsets.all(16.0),
                color: const Color(0xFFEAD8C9), // Background color
                child: Row(
                  mainAxisAlignment: MainAxisAlignment.spaceBetween,
                  children: [
                    const Text(
                      'พนักงาน',
                      style: TextStyle(
                        fontSize: 24,
                        fontWeight: FontWeight.bold,
                      ),
                    ),
                    Image.asset(
                      'assets/images/KobBlur.png', // Adjust the path if needed
                      width: 60,
                      height: 60,
                      fit: BoxFit.cover,
                    )
                  ],
                ),
              ),
              const SizedBox(height: 20),
              // Information Container
              Container(
                padding: const EdgeInsets.all(16.0),
                decoration: BoxDecoration(
                  color: const Color(0xFFD9BA9F),
                  borderRadius: BorderRadius.circular(10),
                  border: Border.all(color: Colors.grey, width: 1),
                  boxShadow: [
                    BoxShadow(
                      color: Colors.grey.withOpacity(0.3),
                      spreadRadius: 5,
                      blurRadius: 10,
                    ),
                  ],
                ),
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      children: [
                        const Text('วันที่:', style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold)),
                        Text(currentDateTime, style: const TextStyle(fontSize: 16)),
                      ],
                    ),
                    const SizedBox(height: 10),
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      children: [
                        const Text('รหัสพนักงาน:', style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold)),
                        Text(employeeId, style: const TextStyle(fontSize: 16)),
                      ],
                    ),
                    const SizedBox(height: 10),
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      children: [
                        const Text('เวลาเข้าทำงาน:', style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold)),
                        Text(checkInTime, style: const TextStyle(fontSize: 16)),
                      ],
                    ),
                    const SizedBox(height: 10),
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      children: [
                        const Text('เวลาออกงาน:', style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold)),
                        Text(checkOutTime, style: const TextStyle(fontSize: 16)),
                      ],
                    ),
                    const SizedBox(height: 10),
                    Row(
                      mainAxisAlignment: MainAxisAlignment.spaceBetween,
                      children: [
                        const Text('สถานะ:', style: TextStyle(fontSize: 16, fontWeight: FontWeight.bold)),
                        Text(workStatus, style: const TextStyle(fontSize: 16)),
                      ],
                    ),
                  ],
                ),
              ),
              const SizedBox(height: 30),
              // Slide Action for Check In/Out
              Container(
                margin: const EdgeInsets.only(top: 32),
                child: SlideAction(
                  text: isCheckedIn ? 'Slide to Check Out' : 'Slide to Check In',
                  textStyle: const TextStyle(color: Colors.black, fontSize: 16, fontWeight: FontWeight.bold),
                  onSubmit: toggleCheckInOut,
                  innerColor: isCheckedIn ? Colors.red : Colors.green,
                  outerColor: Colors.white,
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }
}
